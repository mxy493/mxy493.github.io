 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}

class Solution {

        ListNode *newHead = NULL;   //цЦ░х╗║ф╕Аф╕кщУ╛шбия╝Мх░ЖхОЯщУ╛шбихПНш╜м

        {

        }
        return true;    //щ╗Шшодш┐ФхЫЮtrue
    }
};
