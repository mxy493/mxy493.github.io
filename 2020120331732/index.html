from PySide2.QtCore import QTimer
from PySide2.QtGui import QColor, QPainter, Qt
from PySide2.QtWidgets import QMainWindow, QApplication, QPushButton, QWidget, QVBoxLayout


class CircleLoadingAnimation(QWidget):
    """Âä†ËΩΩÂä®Áîª
    progress = CircleProgressBar(600)
    progress.start()  # ÂºÄÂßãÊòæÁ§∫Âä†ËΩΩÂä®Áîª
    # do something ...
    progress.stop()  # ÂÅúÊ≠¢ÊòæÁ§∫
    """
    def __init__(self, size=100, color=QColor(24, 189, 155), clockwise=True):
        super().__init__()
        self.setAttribute(Qt.WA_TranslucentBackground)  # ÈÄèÊòéÊïàÊûú
        self.setWindowFlags(Qt.FramelessWindowHint  # Êó†ËæπÊ°Ü
                            | Qt.BypassWindowManagerHint
                            | Qt.Tool  # ÈöèÁà∂Á™óÂè£ÂÖ≥Èó≠
                            | Qt.WindowStaysOnTopHint)  # ÁΩÆÈ°∂
        self.setFixedSize(size, size)

        self.angle = 0
        self.clockwise = clockwise  # È°∫Êó∂ÈíàÊñπÂêë
        self.Color = color  # ÂúÜÂúàÈ¢úËâ≤

        self.delta = 36
        self._timer = QTimer(self, timeout=self.update)  # ËÆ°Êó∂Âô®ÔºåÂÆöÊó∂Âà∑Êñ∞ÁïåÈù¢

    def start(self):
        self._timer.start(100)  # 0.1sÊõ¥Êñ∞Âà∑Êñ∞‰∏ÄÊ¨°ÁïåÈù¢
        self.show()  # ÊòæÁ§∫Âä®ÁîªÁ™óÂè£

    def stop(self):
        self._timer.stop()  # ÂÅúÊ≠¢ËÆ°Êó∂Âô®
        self.close()  # ÂÖ≥Èó≠Âä®ÁîªÁ™óÂè£

    def paintEvent(self, event):
        qp = QPainter(self)
        qp.setRenderHint(QPainter.Antialiasing)
        qp.translate(self.width() / 2, self.height() / 2)
        side = min(self.width(), self.height())
        qp.scale(side / 100.0, side / 100.0)
        qp.rotate(self.angle)
        qp.save()
        qp.setPen(Qt.NoPen)
        color = self.Color.toRgb()
        for i in range(11):
            color.setAlphaF(1.0 * i / 10)
            qp.setBrush(color)
            qp.drawEllipse(30, -10, 20, 20)
            qp.rotate(36)
        qp.restore()
        self.angle += self.delta if self.clockwise else -self.delta
        self.angle %= 360


class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle('Âä†ËΩΩÂä®ÁîªDemo')
        self.setFixedSize(400, 400)

        self.progress = CircleLoadingAnimation()  # ÂÆû‰æãÂåñ‰∏Ä‰∏™Âä†ËΩΩÂä®Áîª

        vbox = QVBoxLayout()
        vbox.addWidget(self.btn)
        widget = QWidget()
        widget.setLayout(vbox)
        self.setCentralWidget(widget)

    def show_animation(self):
        if self.progress.isVisible():
            self.progress.stop()
        else:
            self.progress.start()


if __name__ == '__main__':
    app = QApplication(sys.argv)
