 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode* mid = head;//å§‹ç»ˆæŒ‡å‘ä¸­é—´ç»“ç‚¹
        ListNode* cur = head;//å½“å‰ç»“ç‚¹
        int step = 1;//curæ¯éš”ä¸€æ­¥å¾€åç§»ä¸€ä½
        while (cur != NULL) {
            if (step == 0) {

                step = 1;
                cur = cur->next;

            }
            cur = cur->next;
            step--;
        }
        return mid;
    }
};
