class Solution {
public:
    int massage(vector<int>& nums) {
        int n = (int)nums.size();
        if (n == 0) return 0;
        int dp0 = 0, dp1 = nums[0];

        for (int i = 1; i < n; ++i) {
            int tdp0 = max(dp0, dp1); // è®¡ç®— dp[i][0]
            int tdp1 = dp0 + nums[i]; // è®¡ç®— dp[i][1]

            dp0 = tdp0; // ç”¨ dp[i][0] æ›´æ–° dp_0
            dp1 = tdp1; // ç”¨ dp[i][1] æ›´æ–° dp_1
        }
        return max(dp0, dp1);
    }
};
